수업자료 · ML 추론 API
class SpamCheckRequest(BaseModel):
    message: str

class SpamCheckResponse(BaseModel):
    isSpam: bool

@app.post("/api/spam-check",
          response_model=SpamCheckResponse)
async def check_spam(request: SpamCheckRequest):
    model = joblib.load("spam_model.pkl")
    prediction = model.predict([request.message])
    return {"isSpam": bool(prediction[0])}
